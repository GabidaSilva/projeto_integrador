## Turma 33 - Proz Educação - julho a nov/24

Este projeto foi desenvolvido pelo Grupo 01 da Turma 33, no Curso de Introdução à Programação com Foco em Front-End, promovido pela Proz Educação em parceria com a AWS (Talento Cloud). Durante o período de três meses, utilizamos uma abordagem estruturada, com entregas realizadas ao longo de 6 Sprints, para conceber e implementar um blog dedicado a apaixonados por pets.

Com o suporte de ferramentas como o Figma para design, iniciamos o projeto definindo a temática e o layout do blog. Na etapa de desenvolvimento, criamos um repositório no GitHub, implementamos páginas em HTML, estilizações em CSS e adicionamos funcionalidades interativas em JavaScript. Além disso, idealizamos e projetamos o banco de dados com SQLite (utilizando um DER como base), visando uma estrutura eficiente e escalável.

Nosso objetivo principal foi a entrega de um blog funcional e atrativo, que servisse como uma fonte confiável de informações sobre animais de estimação, além de possibilitar oportunidades de monetização futura. Para isso, incorporamos um formulário de captação de parceiros comerciais e um formulário de contato. Também criamos uma calculadora interativa de idade humana de pets, projetada para engajar e entreter os usuários, enquanto destacávamos a relevância informativa do blog.

Entregamos um blog com um design responsivo e as seguintes funcionalidades principais:

6 páginas bem estruturadas, incluindo seções para cães, gatos, parceiros e informações institucionais.
Formulários com validação (Contato e Parcerias), que auxiliam na interação com usuários e na captação de patrocinadores.
Uma calculadora de idade pet que converte a idade dos animais para sua equivalência em idade humana.
Cards interativos com informações detalhadas sobre raças de cães e gatos.
Link para a Política de Privacidade, demonstrando atenção à transparência e conformidade legal.
Responsividade implementada, especialmente na página de parceiros, garantindo uma experiência de navegação otimizada em dispositivos móveis.

## -- criação de tabelas -- SQLite Postgre

CREATE TABLE TipoOrigemUsuario (
 id_origem INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 descricao VARCHAR(50) NOT NULL
);

CREATE TABLE Usuario (
 id_usuario INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 nome VARCHAR(255) NOT NULL,
 email VARCHAR(255) UNIQUE NOT NULL,
 origem INT,
 FOREIGN KEY (origem) REFERENCES TipoOrigemUsuario(id_origem) ON DELETE SET NULL
);

CREATE TABLE Organizacao (
 id_organizacao INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 nome VARCHAR(255) NOT NULL
);
CREATE TABLE TipoParceria (
 id_tipo_parceria INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 descricao VARCHAR(50) NOT NULL
);

CREATE TABLE AreaInteresse (
 id_area INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 descricao VARCHAR(50) NOT NULL
);

CREATE TABLE UsuarioOrganizacao (
 id_usuario INT,
 id_organizacao INT,
 data_associacao DATE DEFAULT CURRENT_DATE,
 PRIMARY KEY (id_usuario, id_organizacao),
 FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) ON DELETE CASCADE,
 FOREIGN KEY (id_organizacao) REFERENCES Organizacao(id_organizacao) ON DELETE 
CASCADE
);

CREATE TABLE Parceiro (
 id_parceiro INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 id_usuario INT,
 id_organizacao INT,
 tipo_parceria INT,
 area_interesse INT,
 contato VARCHAR(255),
 data_inicio DATE,
 data_fim DATE,
 FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) ON DELETE CASCADE,
 FOREIGN KEY (id_organizacao) REFERENCES Organizacao(id_organizacao) ON DELETE SET 
NULL,
 FOREIGN KEY (tipo_parceria) REFERENCES TipoParceria(id_tipo_parceria) ON DELETE SET 
NULL,
 FOREIGN KEY (area_interesse) REFERENCES AreaInteresse(id_area) ON DELETE SET NULL
);

CREATE TABLE Contato (
 id_contato INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 id_usuario INT,
 mensagem TEXT NOT NULL,
 data_envio DATE DEFAULT CURRENT_DATE,
 FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) ON DELETE CASCADE
);

CREATE TABLE Endereco (
 id_endereco INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 id_usuario INT,
 logradouro VARCHAR(255) NOT NULL,
 cidade VARCHAR(100),
 estado VARCHAR(50),
 cep VARCHAR(10),
 tipo_origem INT,
 FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) ON DELETE CASCADE,
 FOREIGN KEY (tipo_origem) REFERENCES TipoOrigemUsuario(id_origem) ON DELETE SET 
NULL
);

CREATE TABLE Telefone (
 id_telefone INT PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY,
 id_usuario INT,
 numero VARCHAR(15) NOT NULL,
 tipo_origem INT,
 FOREIGN KEY (id_usuario) REFERENCES Usuario(id_usuario) ON DELETE CASCADE,
 FOREIGN KEY (tipo_origem) REFERENCES TipoOrigemUsuario(id_origem) ON DELETE SET 
NULL
);

## -- consultas -- 

SELECT nome, email 
FROM Usuario 
WHERE origem = 1; -- Origem 1 corresponde a "Contato"

SELECT nome 
FROM Organizacao 
WHERE nome LIKE '%Amigos%';

SELECT id_parceiro, contato, data_inicio 
FROM Parceiro 
WHERE tipo_parceria = 1; -- Tipo de parceria 1 corresponde a "Colaboração para eventos"

SELECT mensagem, data_envio 
FROM Contato 
WHERE data_envio = '2024-06-10';

SELECT logradouro, cidade, estado 
FROM Endereco 
WHERE cidade = 'São Paulo';

SELECT numero 
FROM Telefone 
WHERE tipo_origem = 2; -- Origem 2 corresponde a "Parceiro"

SELECT U.nome, U.email 
FROM Usuario U
JOIN UsuarioOrganizacao UO ON U.id_usuario = UO.id_usuario
WHERE UO.id_organizacao = 1; -- Organização 1 corresponde ao "Abrigo Esperança"

SELECT U.nome AS usuario, O.nome AS organizacao
FROM Usuario U
INNER JOIN UsuarioOrganizacao UO ON U.id_usuario = UO.id_usuario
INNER JOIN Organizacao O ON UO.id_organizacao = O.id_organizacao;

SELECT P.id_parceiro, U.nome AS usuario, A.descricao AS area_interesse, T.descricao AS 
tipo_parceria
FROM Parceiro P
INNER JOIN Usuario U ON P.id_usuario = U.id_usuario
INNER JOIN AreaInteresse A ON P.area_interesse = A.id_area
INNER JOIN TipoParceria T ON P.tipo_parceria = T.id_tipo_parceria;

SELECT U.nome AS usuario, C.mensagem
FROM Usuario U
LEFT JOIN Contato C ON U.id_usuario = C.id_usuario;

SELECT O.nome AS organizacao, U.nome AS usuario
FROM Organizacao O
LEFT JOIN UsuarioOrganizacao UO ON O.id_organizacao = UO.id_organizacao
LEFT JOIN Usuario U ON UO.id_usuario = U.id_usuario;

SELECT T.descricao AS tipo_parceria, P.id_parceiro, U.nome AS usuario
FROM TipoParceria T
RIGHT JOIN Parceiro P ON T.id_tipo_parceria = P.tipo_parceria
LEFT JOIN Usuario U ON P.id_usuario = U.id_usuario;

SELECT A.descricao AS area_interesse, P.id_parceiro, U.nome AS usuario
FROM AreaInteresse A
RIGHT JOIN Parceiro P ON A.id_area = P.area_interesse
LEFT JOIN Usuario U ON P.id_usuario = U.id_usuario;
